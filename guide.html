2.1. Imports e Parser

import os import sys import time from mip import Model, xsum, minimize, BINARY, CONTINUOUS, OptimizationStatus def ler_steinlib(caminho): with open(caminho, 'r', encoding='utf-8') as f: linhas = f.readlines() n, arestas, terminais, secao = 0, [], [], None for linha in linhas: linha = linha.strip() if not linha or linha.startswith('33D32945'): continue if linha.startswith("SECTION"): secao = linha.split()[1]; continue if linha == "END": continue if secao == "Graph": if linha.startswith("Nodes"): n = int(linha.split()[1]) elif len(linha.split()) >= 3 and linha.split()[0].isdigit(): partes = linha.split() if partes[0] == 'E': partes = partes[1:] u, v, c = int(partes[0]), int(partes[1]), float(partes[2]) arestas.append((u, v, c)) elif secao == "Terminals": if linha.startswith("T"): partes = linha.split() if len(partes) >= 2 and partes[1].isdigit(): terminais.append(int(partes[1])) return n, arestas, terminais

2.2. Construção do Modelo MIP

def construir_e_resolver(n, arestas, terminais): inicio = time.perf_counter() mdl = Model(solver_name='HIGHS') mdl.verbose = 0 y = {e: mdl.add_var(var_type=BINARY) for e in arestas} arcos = [] for u, v, c in arestas: arcos.append((u, v, (u, v, c))) arcos.append((v, u, (u, v, c))) raiz = terminais[0] mercadorias = [t for t in terminais if t != raiz] f = {} for k in mercadorias: for u, v, e in arcos: f[(u, v, k)] = mdl.add_var(var_type=CONTINUOUS, lb=0, ub=1) for k in mercadorias: for i in range(1, n + 1): saida = xsum(f[(i, j, k)] for i_u, j, e in arcos if i_u == i) entrada = xsum(f[(j, i, k)] for j, i_u, e in arcos if i_u == i) if i == raiz: mdl += saida - entrada == 1 elif i == k: mdl += saida - entrada == -1 else: mdl += saida - entrada == 0 for k in mercadorias: for u, v, e in arcos: mdl += f[(u, v, k)] <= y[e] mdl.objective = minimize(xsum(c * y[(u, v, c)] for u, v, c in arestas)) return mdl, inicio
